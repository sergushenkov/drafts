t.task_id,
t.description
from Dwarves d

d.dwarf_id,
d.name,
r.relationship,
d1.dwarf_id as partner_id,
d1.name partner_name,
from Dwarves d
left join Relationships r on d.dwarf_id = r.dwarf_id
left join Dwarfes d1 on d1.dwarf_id = r.related_to;
